import numpy as np

# Создаем матрицу коэффициентов
A = np.array([[1, -1, 1, -4],
              [8, -1, -1, 2],
              [1, 6, -2, -2]])

# Создаем вектор правой части уравнений
b = np.array([-2, 11, -7])

# Применяем метод Гаусса
x = np.linalg.solve(A, b)

# Выводим решение
print("Решение: ")
print("x1 =", x[0])
print("x2 =", x[1])
print("x3 =", x[2])
print("x4 =", x[3])
